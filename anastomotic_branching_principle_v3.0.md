```python
def anastomotic_growth(steps=25, branch_p=0.31, fuse_r=0.09, attractors=None):
    G = nx.Graph()
    pos = {0: np.zeros(3)}
    terminals = [0]
    G.add_node(0)
    
    for _ in range(steps):
        new_terminals = []
        for t in terminals:
            # Branch or extend
            if np.random.rand() < branch_p:
                a, b = len(pos), len(pos)+1
                G.add_edges_from([(t,a),(t,b)])
                direction = np.random.normal(0, 0.12, 3)
                pos[a] = pos[t] + direction
                pos[b] = pos[t] + direction * 1.07
                new_terminals += [a,b]
            else:
                a = len(pos)
                G.add_edge(t, a)
                step = np.random.normal(0, 0.08, 3)
                if attractors:  # optional weak chemotaxis
                    step += 0.03 * (np.random.choice(attractors) - pos[t])
                pos[a] = pos[t] + step
                new_terminals.append(a)
        
        # Anastomosis — the only antifragility step
        fused = set()
        for u,v in combinations(new_terminals, 2):
            if np.linalg.norm(pos[u] - pos[v]) < fuse_r and u not in fused and v not in fused:
                G.add_edge(u, v)      # close loop
                fused.add(u); fused.add(v)
        
        terminals = [t for t in new_terminals if t not in fused]
    
    return G, pos
```
